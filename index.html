        function switchTab(tabName, appType) {

            const prefix = appType === 'admin' ? 'admin-' : 'animator-';

            const app = appType === 'admin' ? 'adminApp' : 'animatorApp';

            

            // Remove active class from all tabs and sections

            document.querySelectorAll(`#${app} .tab`).forEach(t => t.classList.remove('active'));

            document.querySelectorAll(`#${app} .section`).forEach(s => s.classList.remove('active'));

            

            // Add active class to clicked tab and corresponding section

            event.target.classList.add('active');

            const section = document.getElementById(prefix + tabName);

            if (section) section.classList.add('active');

        }


        function loadAdminDashboard() {

            const today = new Date().toISOString().split('T')[0];

            const todayEvents = events.filter(e => e.date === today).length;

            const activeAnimators = workLogs.filter(w => !w.endTime).length;

            const monthHours = workLogs.reduce((total, log) => {

                if (log.duration) return total + log.duration;

                return total;

            }, 0);

            

            document.getElementById('todayEvents').textContent = todayEvents;

            document.getElementById('activeAnimators').textContent = activeAnimators;

            document.getElementById('monthHours').textContent = Math.round(monthHours / 3600000) + 'h';

        }

        function startWork(eventId) {

            if (confirm('Confermi di iniziare il turno?')) {

                currentEventId = eventId;

                workStartTime = new Date();

                

                // Create work log

                const workLog = {

                    id: Date.now().toString(),

                    eventId: eventId,

                    animatorId: currentUser.id,

                    animatorName: currentUser.name,

                    startTime: workStartTime.toISOString(),

                    materials: []

                };

                

                workLogs.push(workLog);

                saveToStorage();

                

                // Update animator status

                const animator = animators.find(a => a.name === currentUser.name);

                if (animator) {

                    animator.status = 'busy';

                    saveToStorage();

                }

                

                // Start timer

                startTimer();

                

                // Switch to active tab

                document.querySelectorAll('#animatorApp .tab').forEach(t => t.classList.remove('active'));

                document.querySelectorAll('#animatorApp .section').forEach(s => s.classList.remove('active'));

                document.querySelector('#animatorApp .tab:nth-child(2)').classList.add('active');

                document.getElementById('animator-active').classList.add('active');

                

                showNotification('Turno Iniziato!', 'Timer avviato. Buon lavoro!');

            }

        }


        function startTimer() {

            workTimer = setInterval(() => {

                const now = new Date();

                const diff = now - workStartTime;

                const hours = Math.floor(diff / 3600000);

                const minutes = Math.floor((diff % 3600000) / 60000);

                const seconds = Math.floor((diff % 60000) / 1000);

                

                document.getElementById('workTimer').textContent = 

                    `${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;

            }, 1000);

        }


        function addMaterial() {

            const input = document.getElementById('materialCode');

            const code = input.value.trim();

            

            if (materialDatabase[code]) {

                if (!materials.find(m => m.code === code)) {

                    materials.push({ code, name: materialDatabase[code] });

                    updateMaterialsList();

                    input.value = '';

                    showNotification('Materiale Aggiunto', materialDatabase[code]);

                }

            } else {

                showNotification('Errore', 'Codice materiale non valido');

            }

        }


        function updateMaterialsList() {

            const list = document.getElementById('materialsList');

            list.innerHTML = materials.map(m => `

        function completeWork() {

            if (!workStartTime) {

                showNotification('Errore', 'Nessun turno attivo');

                return;

            }

            

            if (confirm('Confermi di aver completato il turno e rimesso tutto in magazzino?')) {

                clearInterval(workTimer);

                const endTime = new Date();

                const duration = endTime - workStartTime;

                

                // Update work log

                const currentLog = workLogs.find(w => 

                    w.animatorId === currentUser.id && !w.endTime

                );

                

                if (currentLog) {

                    currentLog.endTime = endTime.toISOString();

                    currentLog.duration = duration;

                    currentLog.materials = [...materials];

                    saveToStorage();

                }

                

                // Update animator status

                const animator = animators.find(a => a.name === currentUser.name);

                if (animator) {

                    animator.status = 'available';

                    saveToStorage();

                }

                

                const hours = Math.floor(duration / 3600000);

                const minutes = Math.floor((duration % 3600000) / 60000);

                

                showNotification('Turno Completato!', `Durata: ${hours}h ${minutes}min`);

                

                // Reset

                workStartTime = null;

                currentEventId = null;

                materials = [];

                document.getElementById('workTimer').textContent = '00:00:00';

                document.getElementById('materialsList').innerHTML = '';

                

                // Switch to history tab

                loadWorkHistory();

                document.querySelectorAll('#animatorApp .tab').forEach(t => t.classList.remove('active'));

                document.querySelectorAll('#animatorApp .section').forEach(s => s.classList.remove('active'));

                document.querySelector('#animatorApp .tab:nth-child(3)').classList.add('active');

                document.getElementById('animator-history').classList.add('active');

            }

        }
